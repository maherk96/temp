```sql
EXPLAIN PLAN FOR
SELECT tg.NAME AS tag, COUNT(DISTINCT t.ID)
FROM TEST_RUN tr
INNER JOIN TEST t ON tr.TEST_ID = t.ID
INNER JOIN TEST_CLASS tc ON t.TEST_CLASS_ID = tc.ID
INNER JOIN APPLICATION a ON tc.APP_ID = a.ID
INNER JOIN TEST_TAG tt ON tt.TEST_LAUNCH_ID = tr.TEST_LAUNCH_ID
INNER JOIN TAG tg ON tg.ID = tt.TAG_ID
WHERE tr.START_TIME > TRUNC(SYSDATE) - 14
  AND a.NAME = 'Fusion Algo'
GROUP BY tg.NAME;

SELECT * FROM TABLE(DBMS_XPLAN.DISPLAY);
```
